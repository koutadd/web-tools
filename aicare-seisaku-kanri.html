const SS_ID         = '1Qe7Ri8rm_GQ21u5m6s533p0R9qAIKqbcBHP5HmLoLf0';
const SHEET_NAME    = '店舗製作物'; // タブ名（製作物）

function doPost(e) {
  let result;
  try {
    const p = JSON.parse(e.postData.contents);
    if (p.action === 'uploadReference') {
      result = uploadReference(p.store, p.item, p.fileName, p.mimeType, p.data);
    } else {
      result = { error: 'unknown post action: ' + p.action };
    }
  } catch(err) {
    result = { error: err.message };
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const p = e.parameter;
  let result;
  try {
    if (p.action === 'getItems')         result = getItems();
    else if (p.action === 'updateStatus')  result = updateStatus(+p.row, p.status);
    else if (p.action === 'createFolders') result = createFolders(p.store, p.item, p.storeNo || '', p.itemNo || '01');
    else if (p.action === 'postToSlack')   result = postToSlack(p.token, p.channelId, p.storeName, p.itemName, p.posterName, p.mentionIds);
    else if (p.action === 'updateProducer')  result = updateProducer(+p.row, p.name);
    else if (p.action === 'getComments')     result = getComments(p.token, p.channelId, p.threadTs);
    else if (p.action === 'postComment')     result = postComment(p.token, p.channelId, p.threadTs, p.text);
    else if (p.action === 'notifyProducer')  result = notifyProducer(p.producerJson, p.message);
    else if (p.action === 'storeSettings')   result = storeSettings(p.json);
    else if (p.action === 'getGallery')         result = getGallery();
    else if (p.action === 'getSlackMembers')    result = getSlackMembers(p.token);
    else if (p.action === 'postChatworkComment') result = postChatworkComment(p.cwToken, p.cwRoomId, p.text, p.posterName);
    else if (p.action === 'postDelivery')       result = postDelivery(p.token, p.channelId, p.store, p.storeNo, p.item, p.folderUrl, p.mentionId);
    else if (p.action === 'postRequest')      result = postRequest(p.token, p.channelId, p.store, p.storeNo, p.item, p.size, p.format, p.requester, p.requestDate, p.deadline, p.notes, p.urgent, p.mentionId);
    else if (p.action === 'addRequest')       result = addRequest(p.store, p.item, p.size, p.notes, p.format, p.requester, p.requestDate, p.deadline, p.urgent);
    else if (p.action === 'getAiSuggestion')  result = getAiSuggestion(p.apiKey, p.mediaType, p.notes);
    else if (p.action === 'getRefImages')     result = getRefImages(p.store, p.storeNo || '');
    else if (p.action === 'getRefImageFull')  result = getRefImageFull(p.id);
    else if (p.action === 'batchMarkComplete') result = batchMarkComplete();
    else result = { error: 'unknown action: ' + p.action };
  } catch(err) {
    result = { error: err.message };
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function getSheet() {
  const ss = SpreadsheetApp.openById(SS_ID);
  const sh = ss.getSheetByName(SHEET_NAME);
  if (sh) return sh;
  const names = ss.getSheets().map(s => '"' + s.getName() + '"').join(', ');
  throw new Error('シートが見つかりません: ' + names);
}

function getItems() {
  ensureDailyReminder();
  const sheet  = getSheet();
  const values = sheet.getDataRange().getValues();
  const sheetGid = sheet.getSheetId();
  // 2パス: H=1でブロック開始、H=3でG列から店舗名取得
  const blocks = [];
  for (let i = 0; i < values.length; i++) {
    const h = String(values[i][7] || '').trim();
    const g = String(values[i][6] || '').trim();
    if (h === '1') {
      if (blocks.length > 0) blocks[blocks.length-1].end = i-1;
      const fallbackName = String(values[i][5] || '').trim(); // F列: 店舗表示名をフォールバックに
      blocks.push({ start:i, end:-1, store: fallbackName, storeNo:String(values[i][3]||''), storeComplete:String(values[i][0]||'').trim() });
    }
    if (h === '3' && g && blocks.length > 0) blocks[blocks.length-1].store = g; // 正式名で上書き
  }
  if (blocks.length > 0) blocks[blocks.length-1].end = values.length-1;

  const items = [];
  for (const b of blocks) {
    if (!b.store) continue;
    for (let i = b.start; i <= b.end; i++) {
      const r    = values[i];
      const iCol = String(r[8]  || '').trim(); // I: 制作アイテム
      if (!iCol) continue;
      items.push({
        rowNum:      i+1,
        store:       b.store,
        storeNo:     b.storeNo,
        item:        iCol,
        size:        String(r[9]  || '').trim(), // J: サイズ
        notes:       String(r[10] || '').trim(), // K: 要望
        format:      String(r[11] || '').trim(), // L: 納品形式
        requester:   String(r[12] || '').trim(), // M: 依頼者
        requestDate: String(r[13] || '').trim(), // N: 依頼日
        deadline:    String(r[14] || '').trim(), // O: 納期
        producer:    String(r[15] || '').trim(), // P: 製作者
        status:      String(r[16] || '').trim() || (b.storeComplete === '完了' ? '完了' : ''), // Q: ステータス（A列=完了の場合フォールバック）
        urgent:      String(r[18] || '').trim(), // S: 急きもの
        cellUrl:     'https://docs.google.com/spreadsheets/d/' + SS_ID + '/edit#gid=' + sheetGid + '&range=G' + (i+1),
      });
    }
  }
  return { items, sheetGid };
}

function updateStatus(rowNum, status) {
  getSheet().getRange(rowNum, 17).setValue(status); // Q列=17番目
  return { ok: true };
}

function batchMarkComplete() {
  const ss = SpreadsheetApp.openById(SS_ID);
  // 店舗マスターシート(gid=471267220)からD列にURLがある店舗Noを収集
  const dirSheet = ss.getSheets().find(s => s.getSheetId() === 471267220);
  if (!dirSheet) return { ok: false, error: '店舗マスターシートが見つかりません (gid=471267220)' };
  const dirValues = dirSheet.getDataRange().getValues();
  const doneNos = new Set();
  for (let i = 1; i < dirValues.length; i++) {
    const no  = String(dirValues[i][1] || '').trim(); // B列: 店舗No
    const url = String(dirValues[i][3] || '').trim(); // D列: フォルダURL
    if (no && url && url.startsWith('http')) doneNos.add(no);
  }
  // 店舗製作物シートのQ列を更新
  const prodSheet = getSheet();
  const values = prodSheet.getDataRange().getValues();
  let currentNo = '';
  let count = 0;
  for (let i = 0; i < values.length; i++) {
    const h = String(values[i][7] || '').trim(); // H列: 通し番号
    if (h === '1') currentNo = String(values[i][3] || '').trim(); // D列: 店舗No
    const item   = String(values[i][8]  || '').trim(); // I列: 制作アイテム
    const status = String(values[i][16] || '').trim(); // Q列: ステータス
    if (item && !status && currentNo && doneNos.has(currentNo)) {
      prodSheet.getRange(i + 1, 17).setValue('完了');
      count++;
    }
  }
  return { ok: true, count };
}

function createFolders(storeName, itemName, storeNo, itemNo) {
  if (!storeNo) {
    return { error: 'storeNo is required: 店舗番号が指定されていません' };
  }
  var shortName = String(storeName || '').replace(/^アイケアLaBo/, '');
  var canonicalStoreName = String(storeNo) + '_' + shortName;
  var mediaFolderName = (itemNo || '01') + '_' + itemName;
  var parent = DriveApp.getFolderById(PARENT_FOLDER);
  var sf = findOrCreateStoreFolder(parent, storeName, shortName, canonicalStoreName);
  if (sf.getName() !== canonicalStoreName) sf.setName(canonicalStoreName);
  var mf   = getOrCreate(sf, mediaFolderName);
  var outf = getOrCreate(mf, 'アウトプット');
  var nd   = getOrCreate(mf, '入稿データ');
  var nm   = getOrCreate(mf, '入稿前データ');
  return {
    ok: true,
    storeFolderName: canonicalStoreName,
    itemFolderName:  mediaFolderName,
    storeFolderUrl:  sf.getUrl(),
    itemFolderUrl:   mf.getUrl(),
    outputFolderUrl: outf.getUrl(),
    nyukouDataUrl:   nd.getUrl(),
    nyukouMaeUrl:    nm.getUrl(),
  };
}

function findOrCreateStoreFolder(parent, storeName, shortName, canonical) {
  var suffix = '_' + storeName;
  var key = normalizeStoreName(shortName);
  var all = parent.getFolders();
  var canonicalMatch = null, numberedMatch = null, legacyMatch = null, looseMatch = null;
  while (all.hasNext()) {
    var f = all.next();
    var n = f.getName();
    if (n === canonical) { canonicalMatch = f; break; }
    if (normalizeStoreName(n) === key) {
      if (/^\d{1,4}(_\d{1,3})?_/.test(n)) {
        if (!numberedMatch) numberedMatch = f;
      } else if (!looseMatch) {
        looseMatch = f;
      }
    }
    if (n === storeName || (suffix && n.endsWith(suffix))) {
      if (!legacyMatch) legacyMatch = f;
    }
  }
  if (canonicalMatch) return canonicalMatch;
  if (numberedMatch)  return numberedMatch;
  if (legacyMatch)    return legacyMatch;
  if (looseMatch)     return looseMatch;
  return parent.createFolder(canonical);
}

// 店舗名/フォルダ名の正規化（番号prefix・店・空白・アイケアLaBo接頭辞を吸収）
function normalizeStoreName(s) {
  return String(s == null ? '' : s)
    .replace(/　/g, '')
    .replace(/\s+/g, '')
    .replace(/^(?:アイケア(?:LaBo|ラボ))/i, '')
    .replace(/^\d{1,4}(_\d{1,3})?_/, '')
    .replace(/店$/, '')
    .toLowerCase();
}
// 親フォルダから店舗フォルダを探す。
// 1) 既存ロジック（完全一致 / _storeName 終端）を最優先で温存
// 2) 見つからなければ正規化キー一致で番号付き正準フォルダを優先採用
// 3) それも無ければ正規化キー一致の裸フォルダ
// 4) どれも無ければ従来通り storeName で新規作成
function findStoreFolder(parent, storeName) {
  const suffix = '_' + storeName;
  const key = normalizeStoreName(storeName);
  const all = parent.getFolders();
  let legacyMatch   = null;
  let numberedMatch = null;
  let looseMatch    = null;
  while (all.hasNext()) {
    const f = all.next();
    const n = f.getName();
    if (n === storeName || n.endsWith(suffix)) {
      if (!legacyMatch) legacyMatch = f;
    }
    if (normalizeStoreName(n) === key) {
      if (/^\d{1,4}(_\d{1,3})?_/.test(n)) {
        if (!numberedMatch) numberedMatch = f;
      } else if (!looseMatch) {
        looseMatch = f;
      }
    }
  }
  if (numberedMatch) return numberedMatch;
  if (legacyMatch)   return legacyMatch;
  if (looseMatch)    return looseMatch;
  return parent.createFolder(storeName);
}

function getOrCreate(parent, name) {
  const it = parent.getFoldersByName(name);
  return it.hasNext() ? it.next() : parent.createFolder(name);
}

function postDelivery(token, channelId, store, storeNo, item, folderUrl, mentionId) {
  var histRes = JSON.parse(UrlFetchApp.fetch(
    'https://slack.com/api/conversations.history?channel=' + channelId + '&limit=200',
    { headers: { Authorization: 'Bearer ' + token } }
  ).getContentText());
  if (!histRes.ok) throw new Error('history: ' + histRes.error);
  var threadTs = null;
  var msgs = histRes.messages || [];
  for (var i = 0; i < msgs.length; i++) {
    if ((msgs[i].text || '').trim() === store) { threadTs = msgs[i].ts; break; }
  }
  if (!threadTs) {
    var parent = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
      method: 'post', contentType: 'application/json',
      headers: { Authorization: 'Bearer ' + token },
      payload: JSON.stringify({ channel: channelId, text: store }),
    }).getContentText());
    if (!parent.ok) throw new Error('parent: ' + parent.error);
    threadTs = parent.ts;
  }
  var atMark  = '@';
  var mention = mentionId
    ? String(mentionId).split(',').map(function(id){ return id.trim(); }).filter(Boolean)
        .map(function(id){ return '<' + atMark + id + '>'; }).join(' ') + ' '
    : '';
  var label   = (storeNo ? storeNo + '_' : '') + store + '_' + item;
  var text    = mention + label + '\nこちら納品いたします！\n' + folderUrl;
  var reply = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
    method: 'post', contentType: 'application/json',
    headers: { Authorization: 'Bearer ' + token },
    payload: JSON.stringify({ channel: channelId, thread_ts: threadTs, text: text }),
  }).getContentText());
  if (!reply.ok) throw new Error('reply: ' + reply.error);
  return { ok: true, ts: reply.ts, threadTs: threadTs };
}

function getGallery() {
  const parent = DriveApp.getFolderById(PARENT_FOLDER);
  const storeIter = parent.getFolders();
  const gallery = [];
  while (storeIter.hasNext()) {
    const sf = storeIter.next();
    const storeName = sf.getName();
    const itemIter = sf.getFolders();
    while (itemIter.hasNext()) {
      const itf = itemIter.next();
      const ifName = itf.getName();
      const item = ifName.startsWith(storeName + '_') ? ifName.slice(storeName.length + 1) : ifName;
      const outIter = itf.getFoldersByName('アウトプット');
      if (!outIter.hasNext()) continue;
      const outf = outIter.next();
      const pdfIter = outf.getFilesByType(MimeType.PDF);
      const pdfs = [];
      while (pdfIter.hasNext()) {
        const f = pdfIter.next();
        pdfs.push({ id: f.getId(), name: f.getName(), url: f.getUrl(), updatedAt: f.getLastUpdated().getTime() });
      }
      if (!pdfs.length) continue;
      gallery.push({ store: storeName, item, folderUrl: itf.getUrl(), pdfs });
    }
  }
  return { gallery };
}

function ensureDailyReminder() {
  try {
    var props = PropertiesService.getScriptProperties();
    if (props.getProperty('reminderSetup') === '1') return;
    ScriptApp.getProjectTriggers().forEach(function(t) {
      if (t.getHandlerFunction() === 'runDailyCheck') ScriptApp.deleteTrigger(t);
    });
    ScriptApp.newTrigger('runDailyCheck').timeBased().everyDays(1).atHour(9).create();
    props.setProperty('reminderSetup', '1');
  } catch(e) { /* 権限なしの場合は無視 */ }
}

function postToSlack(token, channelId, storeName, itemName, posterName, mentionIds) {
  // メンション文字列を構築
  var atMark2 = '@';
  const mentions = (mentionIds || '').split(',').filter(Boolean)
    .map(id => '<' + atMark2 + id.trim() + '>').join(' ');
  const msg = mentions
    ? mentions + ' これから、' + storeName + '_' + itemName + 'に取り掛かります。'
    : 'これから、' + storeName + '_' + itemName + 'に取り掛かります。';

  // 既存スレッド（親メッセージ）を検索
  var histRes = JSON.parse(UrlFetchApp.fetch(
    'https://slack.com/api/conversations.history?channel=' + channelId + '&limit=200',
    { headers: { Authorization: 'Bearer ' + token } }
  ).getContentText());
  if (!histRes.ok) throw new Error('Slack history error: ' + histRes.error);

  var threadTs = null;
  var msgs = histRes.messages || [];
  for (var i = 0; i < msgs.length; i++) {
    if ((msgs[i].text || '').trim() === storeName) {
      threadTs = msgs[i].ts;
      break;
    }
  }

  // なければ親メッセージを新規作成
  if (!threadTs) {
    var newMsg = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
      method: 'post',
      contentType: 'application/json',
      headers: { Authorization: 'Bearer ' + token },
      payload: JSON.stringify({ channel: channelId, text: storeName }),
    }).getContentText());
    if (!newMsg.ok) throw new Error('Slack post parent error: ' + newMsg.error);
    threadTs = newMsg.ts;
  }

  // スレッドにリプライ
  var reply = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
    method: 'post',
    contentType: 'application/json',
    headers: { Authorization: 'Bearer ' + token },
    payload: JSON.stringify({ channel: channelId, thread_ts: threadTs, text: msg }),
  }).getContentText());
  if (!reply.ok) throw new Error('Slack reply error: ' + reply.error);

  return { ok: true, ts: reply.ts, threadTs: threadTs };
}

function getComments(token, channelId, threadTs) {
  var res = JSON.parse(UrlFetchApp.fetch(
    'https://slack.com/api/conversations.replies?channel=' + encodeURIComponent(channelId) + '&ts=' + encodeURIComponent(threadTs),
    { headers: { Authorization: 'Bearer ' + token } }
  ).getContentText());
  if (!res.ok) throw new Error('Slack replies: ' + res.error);
  var msgs = (res.messages || []).slice(1); // 親メッセージを除く
  return { messages: msgs.map(function(m) {
    return { text: m.text, username: m.username || m.user || '不明', ts: m.ts };
  })};
}

function postComment(token, channelId, threadTs, text, posterName) {
  var payload = { channel: channelId, thread_ts: threadTs, text: text };
  if (posterName) { payload.username = posterName; payload.icon_emoji = ':speech_balloon:'; }
  var res = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
    method: 'post', contentType: 'application/json',
    headers: { Authorization: 'Bearer ' + token },
    payload: JSON.stringify(payload),
  }).getContentText());
  if (!res.ok) throw new Error('Slack post: ' + res.error);
  return { ok: true, ts: res.ts };
}

function getSlackMembers(token) {
  var res = JSON.parse(UrlFetchApp.fetch(
    'https://slack.com/api/users.list?limit=200',
    { headers: { Authorization: 'Bearer ' + token } }
  ).getContentText());
  if (!res.ok) throw new Error('Slack users.list: ' + res.error);
  var members = (res.members || [])
    .filter(function(m) { return !m.deleted && !m.is_bot && m.id !== 'USLACKBOT'; })
    .map(function(m) {
      return { id: m.id, name: m.name, displayName: (m.profile && m.profile.display_name) || '', realName: m.real_name || '' };
    });
  return { members: members };
}

function postChatworkComment(cwToken, cwRoomId, text, posterName) {
  var body = posterName ? '[info][title]' + posterName + '[/title]' + text + '[/info]' : text;
  UrlFetchApp.fetch('https://api.chatwork.com/v2/rooms/' + cwRoomId + '/messages', {
    method: 'post',
    headers: { 'X-ChatWorkToken': cwToken },
    payload: { body: body },
  });
  return { ok: true };
}

function notifyProducer(producerJson, message) {
  var p = JSON.parse(producerJson);
  var results = [];
  // Slack DM
  if ((p.notifyVia === 'slack' || p.notifyVia === 'both') && p.slackToken && p.slackUserId) {
    try {
      var dm = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/conversations.open', {
        method: 'post', contentType: 'application/json',
        headers: { Authorization: 'Bearer ' + p.slackToken },
        payload: JSON.stringify({ users: p.slackUserId }),
      }).getContentText());
      if (dm.ok) {
        var pr = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
          method: 'post', contentType: 'application/json',
          headers: { Authorization: 'Bearer ' + p.slackToken },
          payload: JSON.stringify({ channel: dm.channel.id, text: message }),
        }).getContentText());
        results.push({ type: 'slack', ok: pr.ok });
      }
    } catch(e) { results.push({ type: 'slack', ok: false, error: e.message }); }
  }
  // Chatwork
  if ((p.notifyVia === 'chatwork' || p.notifyVia === 'both') && p.chatworkToken && p.chatworkRoomId) {
    try {
      UrlFetchApp.fetch('https://api.chatwork.com/v2/rooms/' + p.chatworkRoomId + '/messages', {
        method: 'post',
        headers: { 'X-ChatWorkToken': p.chatworkToken },
        payload: { body: message },
      });
      results.push({ type: 'chatwork', ok: true });
    } catch(e) { results.push({ type: 'chatwork', ok: false, error: e.message }); }
  }
  return { ok: true, results: results };
}

function runDailyCheck() {
  var props = PropertiesService.getScriptProperties();
  var settingsStr = props.getProperty('slackSettings');
  if (!settingsStr) return;
  var settings;
  try { settings = JSON.parse(settingsStr); } catch(e) { return; }
  var items = getItems().items;
  var today = new Date(); today.setHours(0,0,0,0);
  items.forEach(function(item) {
    if (item.status === '完了' || !item.deadline || !item.producer) return;
    var dl = parseGasDate(item.deadline);
    if (!dl) return;
    var diff = Math.floor((dl - today) / 86400000);
    if ([0,1,3,5].indexOf(diff) < 0) return;
    var producer = (settings.accounts || []).find(function(a) { return a.name === item.producer; });
    if (!producer) return;
    var labels = {0:'【本日締切！】', 1:'【明日締切】', 3:'【締切3日前】', 5:'【締切5日前】'};
    var msg = labels[diff] + '\n' + item.store + '_' + item.item + '\n締切: ' + item.deadline + '\nステータス: ' + (item.status || '未設定');
    try { notifyProducer(JSON.stringify(producer), msg); } catch(e) { Logger.log(e); }
  });
}

function parseGasDate(s) {
  if (!s) return null;
  var m = String(s).match(/(\d{4})[\/\-](\d{1,2})[\/\-](\d{1,2})/);
  if (m) return new Date(+m[1], +m[2]-1, +m[3]);
  m = String(s).match(/^(\d{1,2})\/(\d{1,2})$/);
  if (m) return new Date(new Date().getFullYear(), +m[1]-1, +m[2]);
  var d = new Date(s);
  return isNaN(d.getTime()) ? null : d;
}

function postRequest(token, channelId, store, storeNo, item, size, format, requester, requestDate, deadline, notes, urgent, mentionId) {
  var histRes = JSON.parse(UrlFetchApp.fetch(
    'https://slack.com/api/conversations.history?channel=' + channelId + '&limit=200',
    { headers: { Authorization: 'Bearer ' + token } }
  ).getContentText());
  if (!histRes.ok) throw new Error('history: ' + histRes.error);
  var threadTs = null;
  var msgs = histRes.messages || [];
  for (var i = 0; i < msgs.length; i++) {
    if ((msgs[i].text || '').trim() === store) { threadTs = msgs[i].ts; break; }
  }
  if (!threadTs) {
    var par = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
      method: 'post', contentType: 'application/json',
      headers: { Authorization: 'Bearer ' + token },
      payload: JSON.stringify({ channel: channelId, text: store }),
    }).getContentText());
    if (!par.ok) throw new Error('parent: ' + par.error);
    threadTs = par.ts;
  }
  var atMark3 = '@';
  var mention = mentionId ? '<' + atMark3 + mentionId + '>' : '';
  var label = (storeNo ? storeNo + '_' : '') + store + '_' + item;
  var lines = [
    mention,
    '【制作依頼】' + label,
    size      ? 'サイズ: ' + size : '',
    format    ? '納品形式: ' + format : '',
    deadline  ? '納期: ' + deadline : '',
    requester ? '依頼者: ' + requester : '',
    urgent === 'true' ? '[急ぎ] 急ぎ案件' : '',
    notes     ? '要望: ' + notes : '',
  ].filter(Boolean).join('\n');
  var reply = JSON.parse(UrlFetchApp.fetch('https://slack.com/api/chat.postMessage', {
    method: 'post', contentType: 'application/json',
    headers: { Authorization: 'Bearer ' + token },
    payload: JSON.stringify({ channel: channelId, thread_ts: threadTs, text: lines }),
  }).getContentText());
  if (!reply.ok) throw new Error('reply: ' + reply.error);
  return { ok: true, ts: reply.ts, threadTs: threadTs };
}

function addRequest(storeName, itemName, size, notes, format, requester, requestDate, deadline, urgent) {
  var sheet  = getSheet();
  var values = sheet.getDataRange().getValues();
  var storeGRow = -1;
  var fallbackRow = -1;
  for (var i = 0; i < values.length; i++) {
    var h = String(values[i][7] || '').trim();
    var g = String(values[i][6] || '').trim();
    if (h === '3' && g === storeName) { storeGRow = i; break; }
    if (h === '1' && String(values[i][5] || '').trim() === storeName && fallbackRow === -1) fallbackRow = i;
  }
  if (storeGRow === -1) storeGRow = fallbackRow; // G列空の店舗はF列表示名で照合
  if (storeGRow === -1) throw new Error('店舗が見つかりません: ' + storeName);
  var insertAt = values.length + 1;
  for (var j = storeGRow + 1; j < values.length; j++) {
    var hj = String(values[j][7] || '').trim();
    if (hj === '1') { insertAt = j + 1; break; }
  }
  var rowData = new Array(19).fill('');
  rowData[8]  = itemName || '';
  rowData[9]  = size || '';
  rowData[10] = notes || '';
  rowData[11] = format || '';
  rowData[12] = requester || '';
  rowData[13] = requestDate || '';
  rowData[14] = deadline || '';
  rowData[18] = (urgent === 'true') ? '急' : '';
  if (insertAt > values.length) {
    sheet.appendRow(rowData);
    return { ok: true, rowNum: values.length + 1 };
  } else {
    sheet.insertRowBefore(insertAt);
    sheet.getRange(insertAt, 1, 1, 19).setValues([rowData]);
    return { ok: true, rowNum: insertAt };
  }
}

function uploadReference(storeName, itemName, fileName, mimeType, base64data) {
  var parent = DriveApp.getFolderById(PARENT_FOLDER);
  var refFolder = getOrCreate(parent, '参考資料');
  // 店舗フォルダ > アイテムフォルダ
  var storeFolder = getOrCreate(refFolder, storeName || '共通');
  var label = (itemName || '').substring(0, 50);
  var targetFolder = label ? getOrCreate(storeFolder, label) : storeFolder;
  var blob = Utilities.newBlob(Utilities.base64Decode(base64data), mimeType || 'application/octet-stream', fileName);
  var file = targetFolder.createFile(blob);
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  return { ok: true, url: file.getUrl(), id: file.getId() };
}

function getAiSuggestion(apiKey, mediaType, notes) {
  if (!apiKey) throw new Error('API Keyが設定されていません');
  var prompt = 'あなたは印刷・サインデザイン制作の専門アシスタントです。\n媒体: ' + mediaType + '\n要望・参考情報: ' + (notes || '特になし') + '\n\nこの制作物について日本語で簡潔に以下をアドバイスしてください：\n・代表的なサイズ候補（2〜3例）\n・デザインのポイント\n・制作時の注意点';
  var url = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=' + apiKey;
  var res = UrlFetchApp.fetch(url, {
    method: 'post',
    contentType: 'application/json',
    payload: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
  });
  var data = JSON.parse(res.getContentText());
  var cand = (data.candidates || [])[0] || {};
  var text = ((((cand.content || {}).parts || [])[0]) || {}).text || 'AIの返答を取得できませんでした';
  return { ok: true, suggestion: text };
}

// 店舗フォルダ内の画像を集める（モーダル「参考画像（店舗フォルダ）」用）
function getRefImages(storeName, storeNo) {
  if (!storeName) return { ok: true, images: [] };
  var parent = DriveApp.getFolderById(PARENT_FOLDER);
  var shortName = String(storeName || '').replace(/^アイケアLaBo/, '');
  var sf = findStoreFolderOnly_(parent, storeName, shortName);
  if (!sf) return { ok: true, images: [] };
  var images = [];
  var IMG_RE = /\.(jpe?g|png|gif|webp|heic|bmp|tiff?)$/i;
  var seen = {};
  function pushImage(f) {
    var id = f.getId();
    if (seen[id]) return;
    var name = f.getName();
    var mime = f.getMimeType() || '';
    if (!IMG_RE.test(name) && mime.indexOf('image/') !== 0) return;
    seen[id] = true;
    images.push({ id: id, name: name, thumb: 'https://drive.google.com/thumbnail?id=' + id + '&sz=w240', mime: mime });
  }
  var directFiles = sf.getFiles();
  while (directFiles.hasNext() && images.length < 200) pushImage(directFiles.next());
  var preferred = ['参考画像', '参考資料', 'reference', 'Reference', 'ref'];
  for (var i = 0; i < preferred.length; i++) {
    var it = sf.getFoldersByName(preferred[i]);
    while (it.hasNext()) {
      var sub = it.next();
      var ff = sub.getFiles();
      while (ff.hasNext() && images.length < 200) pushImage(ff.next());
    }
  }
  var subFolders = sf.getFolders();
  while (subFolders.hasNext() && images.length < 200) {
    var sub2 = subFolders.next();
    if (preferred.indexOf(sub2.getName()) >= 0) continue;
    for (var j = 0; j < preferred.length; j++) {
      var it2 = sub2.getFoldersByName(preferred[j]);
      while (it2.hasNext() && images.length < 200) {
        var deep = it2.next();
        var dfs = deep.getFiles();
        while (dfs.hasNext() && images.length < 200) pushImage(dfs.next());
      }
    }
  }
  return { ok: true, images: images };
}

function findStoreFolderOnly_(parent, storeName, shortName) {
  var suffix = '_' + storeName;
  var key = normalizeStoreName(shortName);
  var all = parent.getFolders();
  var numberedMatch = null, legacyMatch = null, looseMatch = null;
  while (all.hasNext()) {
    var f = all.next();
    var n = f.getName();
    if (normalizeStoreName(n) === key) {
      if (/^\d{1,4}(_\d{1,3})?_/.test(n)) {
        if (!numberedMatch) numberedMatch = f;
      } else if (!looseMatch) {
        looseMatch = f;
      }
    }
    if (n === storeName || (suffix && n.endsWith(suffix))) {
      if (!legacyMatch) legacyMatch = f;
    }
  }
  return numberedMatch || legacyMatch || looseMatch || null;
}

function getRefImageFull(fileId) {
  if (!fileId) return { error: 'id is required' };
  try {
    var file = DriveApp.getFileById(fileId);
    var mime = file.getMimeType() || 'application/octet-stream';
    if (mime.indexOf('image/') !== 0) return { error: 'not an image: ' + mime };
    var blob = file.getBlob();
    var b64 = Utilities.base64Encode(blob.getBytes());
    return { ok: true, data: 'data:' + mime + ';base64,' + b64, name: file.getName() };
  } catch (err) {
    return { error: err.message };
  }
}
